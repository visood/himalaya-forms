#+begin_src jupyter-python
import numpy as np
from typing import NamedTuple

class BoundingBox(NamedTuple):
    north: float
    south: float
    east: float
    west: float

def create_grid(bbox: BoundingBox, tile_size_degrees: float = 0.5):
    """
    Divide a bounding box into smaller tiles.

    Args:
        bbox: BoundingBox containing north, south, east, west coordinates
        tile_size_degrees: Size of each tile in degrees

    Returns:
        List of BoundingBox objects for each tile
    """
    lat_tiles = np.arange(bbox.south, bbox.north, tile_size_degrees)
    lon_tiles = np.arange(bbox.west, bbox.east, tile_size_degrees)

    tiles = []
    for lat in lat_tiles:
        for lon in lon_tiles:
            tile = BoundingBox(
                north=min(lat + tile_size_degrees, bbox.north),
                south=lat,
                east=min(lon + tile_size_degrees, bbox.east),
                west=lon
            )
            tiles.append(tile)

    return tiles

# Define Himachal Pradesh bounding box
hp_bbox = BoundingBox(

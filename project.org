#+begin_src cpp
#include <GL/glut.h>
#include <vector>
#include <cmath>

// Heightmap data
const int MAP_SIZE = 100;
std::vector<std::vector<float>> heightmap(MAP_SIZE, std::vector<float>(MAP_SIZE));

void initHeightmap() {
    // Generate a simple heightmap (you can load from file instead)
    for(int x = 0; x < MAP_SIZE; x++) {
        for(int z = 0; z < MAP_SIZE; z++) {
            float height = sin(x * 0.1) * cos(z * 0.1) * 5.0;
            heightmap[x][z] = height;
        }
    }
}

void renderTerrain() {
    glBegin(GL_TRIANGLES);
    for(int x = 0; x < MAP_SIZE-1; x++) {
        for(int z = 0; z < MAP_SIZE-1; z++) {
            // Create triangles for each grid cell
            glVertex3f(x, heightmap[x][z], z);
            glVertex3f(x+1, heightmap[x+1][z], z);
            glVertex3f(x, heightmap[x][z+1], z+1);

            glVertex3f(x+1, heightmap[x+1][z], z);
            glVertex3f(x+1, heightmap[x+1][z+1], z+1);
            glVertex3f(x, heightmap[x][z+1], z+1);
        }
    }
    glEnd();
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
    glLoadIdentity();

    // Set camera position
    gluLookAt(50.0, 30.0, 50.0,  // Eye position
              50.0, 0.0, 50.0,   // Look at point
              0.0, 1.0, 0.0);    // Up vector

    renderTerrain();
    glutSwapBuffers();
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB | GLUT_DEPTH);
    glutInitWindowSize(800, 600);
    glutCreateWindow("Terrain Visualization");

    initHeightmap();

    glutDisplayFunc(display);
    glEnable(GL_DEPTH_TEST);

    glutMainLoop();
    return 0;
}
